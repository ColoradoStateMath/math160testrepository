\begin{problem}
    What is the circumference and area of a circle with diameter $7.5$?  \it{Answers are in the form (circumference, area)}
    
    \begin{hint}
    If you are having trouble remembering how to calculate the circumference or area of a circle, \href{https://ximera.osu.edu/math160fa17/m160prerequisites/prerequisiteVideos/trigGeometricMeasures}{this card} may help jog your memory.
    \end{hint}
   

  \begin{multipleChoice}
      \choice{$\left( 15\pi, \pi(7.5)^2 \right)$}
      \choice{$\left( 7.5\pi, 7.5\pi^2 \right)$}
      \choice{$\left( 3.75\pi, \pi(7.5)^2 \right)$}
      \choice[correct]{$\left( 7.5\pi, \pi(3.75)^2 \right)$}
      \choice{$\left( \pi(3.75)^2, 3.75\pi \right)$}
      \choice{$\left( \pi(7.5)^2, 15\pi \right)$}
      \choice{$\left( \pi(7.5)^2, 7.5\pi \right)$}
      \choice{$\left( \pi(3.75)^2, 7.5\pi \right)$}
      
      
  \end{multipleChoice}
  
\end{problem}

\begin{question}
  
  If you didn't, why didn't you answer the above question?
  
  \begin{multipleChoice}
      \choice[correct]{I don't know the answer to this question yet.}
      \choice[correct]{I don't want to answer this question right now.}
      \choice[correct]{I have already answered this question correctly.}  
      
  \end{multipleChoice}
  
\end{question}
  
\end{problem}
